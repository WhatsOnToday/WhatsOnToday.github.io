177                }

194     *
195     * So this function caches the response from the OAuth provider and returns it on the second
196     * request. Not nice, but it works as a temporary workaround until the bug is fixed.
197     *
198     * @param request The current request.
199     * @param f The action to execute.
200     * @return A result.
201     * @see https://github.com/sahat/satellizer/issues/287
202     */
203   private def cacheAuthTokenForOauth1(request: Request[AnyContent])(f: Request[AnyContent] => Future[Result]): Future[Result] = {
204     request.getQueryString("oauth_token") -> request.getQueryString("oauth_verifier") match {
205       case (Some(token), Some(verifier)) => cache.get[Result](token + "-" + verifier).flatMap {
206         case Some(r) => Future(r)
207         case None => f(request).map { result =>
208           cache.set(token + "-" + verifier, result, 1 minute)
209           result
210         }
211       }
212       case _ => f(request)
213     }
