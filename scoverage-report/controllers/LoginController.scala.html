159   def authenticate(provider: String, userType: Option[String]) = Action.async { r =>
160     cacheAuthTokenForOauth1(r) { implicit request =>
161       (socialProviderRegistry.get[SocialProvider](provider) match {
162          case Some(p: OAuth2Provider with CommonSocialProfileBuilder) =>
163            p.authenticate(UserStateItem(List("type"->userType.getOrElse("user")).toMap)).flatMap {
164              case Left(result) => Future.successful(result)
165              case Right(StatefulAuthInfo(authInfo,state)) => for {
166                profile <- p.retrieveProfile(authInfo)
167                login <- loginService.save(profile, userType.getOrElse("user"), Some(authInfo.accessToken))
168                authInfo <- authInfoRepository.save(profile.loginInfo, authInfo)
169                authenticator <- silhouette.env.authenticatorService.create(profile.loginInfo)
170                token <- silhouette.env.authenticatorService.init(authenticator)
171                avatar <- avatarService.retrieveURL(profile.email.getOrElse(""))
172              } yield {
173                state.state.get("type") match {
174                  case Some("user") => userService.save(login,avatar)
175                  case Some("organizer") => organizerService.save(login,
176                                                                  profile.firstName.getOrElse("") + " " + profile.lastName.getOrElse(""),
177                                                                  avatar)
178                  case _ =>
179                }
180                log.debug(authInfo.toString)
181                silhouette.env.eventBus.publish(LoginEvent(login, request))
182                Redirect(applicationConfig.url + "web?token=" + token + "&userType=" + userType.getOrElse("user"))
183              }
184            }

195   /**
196     * Satellizer executes multiple requests to the same application endpoints for OAuth1.
197     *
198     * So this function caches the response from the OAuth provider and returns it on the second
199     * request. Not nice, but it works as a temporary workaround until the bug is fixed.
200     *
201     * @param request The current request.
202     * @param f The action to execute.
203     * @return A result.
204     * @see https://github.com/sahat/satellizer/issues/287
205     */
206   private def cacheAuthTokenForOauth1(request: Request[AnyContent])(f: Request[AnyContent] => Future[Result]): Future[Result] = {
207     request.getQueryString("oauth_token") -> request.getQueryString("oauth_verifier") match {
208       case (Some(token), Some(verifier)) => cache.get[Result](token + "-" + verifier).flatMap {
209         case Some(r) => Future(r)
210         case None => f(request).map { result =>
211           cache.set(token + "-" + verifier, result, 1 minute)
212           result
213         }
214       }
215       case _ => f(request)
216     }
