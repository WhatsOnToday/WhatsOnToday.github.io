169   /**

171     *
172     * So this function caches the response from the OAuth provider and returns it on the second
173     * request. Not nice, but it works as a temporary workaround until the bug is fixed.
174     *
175     * @param request The current request.
176     * @param f The action to execute.
177     * @return A result.
178     * @see https://github.com/sahat/satellizer/issues/287
179     */
180   private def cacheAuthTokenForOauth1(request: Request[AnyContent])(f: Request[AnyContent] => Future[Result]): Future[Result] = {
181     request.getQueryString("oauth_token") -> request.getQueryString("oauth_verifier") match {
182       case (Some(token), Some(verifier)) => cache.get[Result](token + "-" + verifier).flatMap {
183         case Some(r) => Future(r)
184         case None => f(request).map { result =>
185           cache.set(token + "-" + verifier, result, 1 minute)
186           result
187         }
188       }
189       case _ => f(request)
190     }
