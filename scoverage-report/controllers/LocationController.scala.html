56     log.debug("Rest request to get location close to another location")
57 
58     val q = for(l <- location if l.id === id.bind) yield l;
59 
60     db.run(q.detailed).map(_.headOption).flatMap {
61       case Some(r) => {
62         val q = location.sortBy(y => geoDistance(r.latitude, r.longitude, y.latitude, y.longitude))
