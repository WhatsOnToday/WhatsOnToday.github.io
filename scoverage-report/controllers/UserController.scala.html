59   def signUp = Action.async(parse.json) { implicit request =>
60     UserSignUpForm.form.bindFromRequest.fold(
61       form => {
62         Future.successful(BadRequest(Json.toJson(form.errors)))
63       },
64       data => {
65         val loginInfo = LoginInfo(CredentialsProvider.ID, data.email)
66         loginService.retrieveAll(loginInfo).flatMap {
67           case Some(login) =>
68             Future.successful(BadRequest(Json.obj("message" -> "user.exists")))
69           case None =>
70             val authInfo = passwordHasher.hash(data.password)
71             val login = Login(None, data.email, None, None, None, loginInfo.providerID, loginInfo.providerKey, false, "user")
72             for {
73               login <- loginService.save(login)
74               authInfo <- authInfoRepository.add(loginInfo, authInfo)
75               authenticator <- silhouette.env.authenticatorService.create(loginInfo)
76               token <- silhouette.env.authenticatorService.init(authenticator)
77               avatar <- avatarService.retrieveURL(data.email)
78             } yield {
79               silhouette.env.eventBus.publish(SignUpEvent(login, request))
80               silhouette.env.eventBus.publish(LoginEvent(login, request))
81               userService.save(login,avatar)
82               mailService.sendUserConfirmation(data.email,token)
83               Ok(Json.obj("message" -> "mail.sent"))
84             }
85         }
86       })
87   }
88

97   }
