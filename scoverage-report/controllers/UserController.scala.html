51   def signUp = Action.async(parse.json) { implicit request =>
52     UserSignUpForm.form.bindFromRequest.fold(
53       form => {
54         Future.successful(BadRequest(Json.toJson(form.errors)))
55       },
56       data => {
57         val loginInfo = LoginInfo(CredentialsProvider.ID, data.email)
58         loginService.retrieveAll(loginInfo).flatMap {
59           case Some(login) =>
60             Future.successful(BadRequest(Json.obj("message" -> "user.exists")))
61           case None =>
62             val authInfo = passwordHasher.hash(data.password)
63             val login = Login(None, data.email, None, None, None, loginInfo.providerID, loginInfo.providerKey, false)
64             for {
65               login <- loginService.save(login)
66               authInfo <- authInfoRepository.add(loginInfo, authInfo)
67               authenticator <- silhouette.env.authenticatorService.create(loginInfo)
68               token <- silhouette.env.authenticatorService.init(authenticator)
69               avatar <- avatarService.retrieveURL(data.email)
70             } yield {
71               silhouette.env.eventBus.publish(SignUpEvent(login, request))
72               silhouette.env.eventBus.publish(LoginEvent(login, request))
73               userService.save(login,avatar)
74               mailService.sendUserConfirmation(data.email,token)
75               Ok(Json.obj("message" -> "mail.sent"))
76             }
77         }
78       })
79   }
