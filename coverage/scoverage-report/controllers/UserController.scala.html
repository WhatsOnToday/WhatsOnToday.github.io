50   def signUp = Action.async(parse.json) { implicit request =>
51     UserSignUpForm.form.bindFromRequest.fold(
52       form => {
53         Future.successful(BadRequest(Json.toJson(form.errors)))
54       },
55       data => {
56         val loginInfo = LoginInfo(CredentialsProvider.ID, data.email)
57         loginService.retrieveAll(loginInfo).flatMap {
58           case Some(login) =>
59             Future.successful(BadRequest(Json.obj("message" -> "user.exists")))
60           case None =>
61             val authInfo = passwordHasher.hash(data.password)
62             val login = Login(None, data.email, None, None, None, loginInfo.providerID, loginInfo.providerKey, false)
63             for {
64               login <- loginService.save(login)
65               authInfo <- authInfoRepository.add(loginInfo, authInfo)
66               authenticator <- silhouette.env.authenticatorService.create(loginInfo)
67               token <- silhouette.env.authenticatorService.init(authenticator)
68               avatar <- avatarService.retrieveURL(data.email)
69             } yield {
70               silhouette.env.eventBus.publish(SignUpEvent(login, request))
71               silhouette.env.eventBus.publish(LoginEvent(login, request))
72               userService.save(login,avatar)
73               mailService.sendUserConfirmation(data.email,token)
74               Ok(Json.obj("message" -> "mail.sent"))
75             }
76         }
77       })
78   }
