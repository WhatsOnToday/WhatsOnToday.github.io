137                }

161     * @see https://github.com/sahat/satellizer/issues/287
162     */
163   private def cacheAuthTokenForOauth1(request: Request[AnyContent])(f: Request[AnyContent] => Future[Result]): Future[Result] = {
164     request.getQueryString("oauth_token") -> request.getQueryString("oauth_verifier") match {
165       case (Some(token), Some(verifier)) => cache.get[Result](token + "-" + verifier).flatMap {
166         case Some(r) => Future(r)
167         case None => f(request).map { result =>
168           cache.set(token + "-" + verifier, result, 1 minute)
169           result
170         }
171       }
172       case _ => f(request)
173     }
